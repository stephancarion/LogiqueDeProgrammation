DEBUT moyenne dans un intervalle
	VARIABLES LOCALES
		var cptBoucle : N <- 0
		var cptMoyenne : N <- 0
		var nombre : N <- 0
		var somme : N <- 0
	FIN VARIABLES LOCALES
	
	POUR cptBoucle ALLANT DE 1 A 30
		ECRIRE "Entrez un nombre"
		SAISIR nombre
		
		SI nombre >=10 ET nombre <= 20
			ALORS
				somme <- somme + nombre
				cptMoyenne <- cptMoyenne + 1
		FINSI
	cptBoucle SUIVANT
	SI cptMoyenne > 0
		ALORS
			ECRIRE "Moyenne des nombres entre 10 et 20", somme/cptMoyenne
		SINON
			ECRIRE "Aucun des 30 nombres entr√©s est compris entre 10 et 20"
	FINSI
FIN moyenne dans un intervalle
